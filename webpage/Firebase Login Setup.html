Store Cart in Firebase Firestore

document.querySelectorAll(".add-to-cart").forEach((button) => {
    button.addEventListener("click", (e) => {
        const user = firebase.auth().currentUser;
        if (!user) {
            alert("Please log in to add items to the cart!");
            return;
        }

        const itemName = e.target.getAttribute("data-name");
        const itemPrice = e.target.getAttribute("data-price");

        const cartRef = firebase.firestore().collection("carts").doc(user.uid);

        cartRef.get().then((doc) => {
            let cartItems = doc.exists ? doc.data().items : [];
            cartItems.push({ name: itemName, price: itemPrice });

            cartRef.set({ items: cartItems }).then(() => {
                alert(`${itemName} added to cart!`);
                updateCartCount(cartItems.length);
            });
        });
    });
});

// Update Cart Count in Navbar
function updateCartCount(count) {
    document.getElementById("cart-count").innerText = count;
}
